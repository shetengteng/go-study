# 介绍

- os.File封装所有文件相关操作，File是一个结构体

- 常用方法

```go
func Open(name string) (file *File,err error)
func (f *File) Close() error
```

- 示例

```go
package main
import (
	"os"
	"fmt"
)
func main(){
	file,err := os.Open("d:/test.txt")
	if err != nil {
		fmt.Println("open file err=",err)
	}else {
		fmt.Printf("file=%v",file)
	}
	err = file.Close()
	if err != nil {
		fmt.Println("close file err=",err)
	}
}
// result: file是一个文件指针
file=&{0xc0420785a0}
```



# 缓冲区

- 读取文件的内容并显示在终端(带缓冲区的方式)，使用os.Open,file.Close,bufio.NewReader(),reader.ReadString函数和方法

```go
package main
import (
	"io"
	"bufio"
	"os"
	"fmt"
)
func main(){
	file ,err := os.Open("d:/test.txt")
	if err != nil {
		fmt.Println("err=",err)
	}
	// 当函数退出时，及时关闭file
	defer file.Close()

	// 创建一个 *Reader 带有缓冲，如下是源码中定义的
	// const(
	// 	defaultBufSize = 4096
	// )
	reader := bufio.NewReader(file)
	for{
		str,err:= reader.ReadString('\n')  
		// 读到一个换行就结束
		// 注意如果没有换行表示结束，那么在读取的文件最后需要一个换行表示结束
		if err == io.EOF { // io.EOF 表示文件末尾
			break
		}else{
			fmt.Print(str)
		}
	}
	fmt.Println("文件读取完成")
}
```



# 一次性读取

- 读取文件的内容并显示在终端(使用ioutil一次将整个文件读入到内存中)，这种方式适用于文件不大的情况。相关方法和函数(ioutil.ReadFile)

```go
package main
import (
	"io/ioutil"
	"fmt"
)
func main(){
	file := "d:/test.txt"
	content ,err := ioutil.ReadFile(file)
	if err != nil {
		fmt.Printf("read file err=%v",err)
	}
	// 将读取的内容显示在终端
	fmt.Printf("%v",string(content)) // []byte
	// open文件和close文件都封装在ReadFile函数内部
}
```

- 带缓冲的写入

```go
package main
import (
	"bufio"
	"os"
	"fmt"
)
func main(){

	filePath := "d:/s.txt"
	file,err:=os.OpenFile(filePath,os.O_WRONLY | os.O_CREATE,0666)
	if err != nil {
		fmt.Println("err=",err)
		return
	}
	defer file.Close()
	str := "hello go\n"
	writer := bufio.NewWriter(file)
	for i := 0; i < 5; i++ {
		writer.WriteString(str)
	}
	// 由于writer是带缓存的，在调用WriterString方法时，先写入缓存
	//最后需要flush将缓冲的数据写入到文件中
	writer.Flush()
}
```

# 判断文件是否存在

- 使用os.Stat()函数判断是否存在
  - 返回错误为nil，说明存在
  - 返回错误类型使用os.IsNotExist()判断为true，说明文件或文件夹不存在
  - 如果返回的错误为其他类型，则不确定是否存在

```go
func PathExists(path string)(bool,err){
	_,err := os.Stat(path)
	if err == nil {
		return true,nil
	}
	if os.IsNotExits(err){
		return false,nil
	}
	return false,err
}
```



# 示例：拷贝文件

- 将一张图片/电影/mp3拷贝到另外一个文件e:/abc.jpg
  - 如果文件不存在需要创建该文件
  - 使用io.Copy函数
  - 注意，此处不能使用bufio，会导致数据拷贝不完全，而且io.Copy中已经有了缓冲区

```go
package main
import (
	"io"
	"os"
	"fmt"
)
func main(){
	srcFile := "d:/abc.png"
	dstFile := "d:/abc3.png"
	_,err := CopyFile(dstFile,srcFile)
	if err == nil {
		fmt.Println("finish")
	}else{
		fmt.Println("err=",err)
	}
}	

func CopyFile(dstFileName string,srcFileName string)(written int64,err error){
	srcFile,err := os.OpenFile(srcFileName,os.O_RDWR,0666)
	if err != nil {
		fmt.Printf("open file err=%v \n",err)
		return
	}
	defer srcFile.Close()
	
	// 通过srcfile 获取Reader
	// reader := bufio.NewReader(srcFile)
	reader := io.Reader(srcFile)
	
	// 打开dstFileName
	dstFile,err := os.OpenFile(dstFileName,os.O_WRONLY|os.O_CREATE,0666)
	if err != nil {
		fmt.Println("open file err=",err)
		return
	}
	defer dstFile.Close()
	
	// writer := bufio.NewWriter(dstFile)
	writer := io.Writer(dstFile)
	return io.Copy(writer,reader)
}
```



# 示例：统计不同类型的字符个数

- 统计一个文件中含有的英文、数字、空格及其它字符数量

```go
package main
import (
	"os"
	"bufio"
	"io"
	"fmt"
)

type CharCount struct {
	ChCount int // 记录英文个数
	NumCount int // 记录数字个数
	SpaceCount int // 记录空格个数
	OtherCount int // 记录其他字符的个数
}

func main(){

	// 打开一个文件，创建一个reader
	// 读取每一行，进行统计
	// 将结果保存到一个结构体中
	fileName := "d:/abc.txt"
	file,err := os.Open(fileName)
	if err != nil {
		fmt.Println("err=",err)
		return
	}
	defer file.Close()
	var count CharCount
	reader := bufio.NewReader(file)
	for {
		str,err := reader.ReadString('\n')
		if err == io.EOF{
			break
		}
		arr := []rune(str)
		for _,v := range arr {
			switch  {
			case v >= 'a' && v <='z':
				fallthrough // 穿透
			case v >= 'A' && v <= 'Z':
				count.ChCount ++
			case v == ' ' || v == '\t':
				count.SpaceCount ++
			case v >= '0' && v <= '9':
				count.NumCount ++
			default:
				count.OtherCount ++
			}
		}
	}
	fmt.Println(count)
}
```
